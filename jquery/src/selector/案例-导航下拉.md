js
~~~
$(function(){
	//只要 li（有ul）被hover 让里面的ul 下拉显示
	//只要 li（有ul）被out  让里面的ul 隐藏
	$("li:has(ul)").hover(function(){
		var myul=$(this).children("ul");
		if(myul.is(":animated")){
			myul.stop(true,true);
			// stop(停止到队列动画，并回到最终状态)
		}
		myul.slideToggle()//第一次显示 第二次隐藏，第三次显示
	})
	
	
})
